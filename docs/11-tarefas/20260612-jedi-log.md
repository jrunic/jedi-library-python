### `jedi_library/log.py` (reescrita)

```python
import contextvars
import json
import logging
import os
import socket
import sys
import traceback
from datetime import datetime, UTC

_execution_id: contextvars.ContextVar[str | None] = contextvars.ContextVar(
    "jedi_log_execution_id", default=None,
)

_HOST = socket.gethostname().split(".")[0]

class JsonFormatter(logging.Formatter):
    def __init__(
        self,
        *,
        service: str | None,
        service_version: str | None,
        actor: str,
        actor_kind: str,
    ) -> None:
        super().__init__()
        self._service = service
        self._service_version = service_version
        self._actor = actor
        self._actor_kind = actor_kind

    def format(self, record: logging.LogRecord) -> str:
        ts = datetime.fromtimestamp(record.created, tz=UTC).isoformat().replace("+00:00", "Z")
        out: dict = {
            "ts": ts,
            "level": record.levelname,
            "logger": record.name,
            "msg": record.getMessage(),
            "module": record.module,
            "func": record.funcName,
            "line": record.lineno,
            "pid": record.process,
            "host": _HOST,
        }
        if self._service:
            out["service"] = self._service
        if self._service_version:
            out["service_version"] = self._service_version
        out["actor"] = self._actor
        out["actor_kind"] = self._actor_kind

        exec_id = _execution_id.get()
        if exec_id:
            out["execution_id"] = exec_id

        # Coleta extras (kwargs `extra=...`) em `metadata`, excluindo campos reservados do LogRecord
        reserved = set(logging.LogRecord.__dict__) | {"message", "asctime"}
        metadata = {
            k: v for k, v in record.__dict__.items()
            if k not in reserved and not k.startswith("_")
        }
        if metadata:
            out["metadata"] = metadata

        if record.exc_info:
            exc_type, exc_val, exc_tb = record.exc_info
            out["exc"] = {
                "type": exc_type.__name__ if exc_type else "Unknown",
                "msg": str(exc_val) if exc_val else "",
                "stack": "".join(traceback.format_exception(exc_type, exc_val, exc_tb)),
            }

        return json.dumps(out, ensure_ascii=False, default=str)


def setup_logging(
    *,
    level: str = "INFO",
    service: str | None = None,
    service_version: str | None = None,
    actor: str | None = None,
    actor_kind: str = "human",
) -> None:
    """Configura root logger com JsonFormatter em stdout. Idempotente."""
    resolved_actor = (
        actor
        or os.environ.get("JEDI_LOG_ACTOR")
        or os.environ.get("USER")
        or "unknown"
    )
    resolved_actor_kind = os.environ.get("JEDI_LOG_ACTOR_KIND", actor_kind)

    handler = logging.StreamHandler(sys.stdout)
    handler.setFormatter(JsonFormatter(
        service=service,
        service_version=service_version,
        actor=resolved_actor,
        actor_kind=resolved_actor_kind,
    ))
    root = logging.getLogger()
    root.handlers = [handler]
    root.setLevel(getattr(logging, level.upper()))


def set_execution_id(execution_id: str | None) -> None:
    """Define execution_id da run atual via ContextVar. None remove."""
    _execution_id.set(execution_id)
```

### Testes (`test/python/test_log.py`)

Cobertura mínima:
- Output é JSON válido em uma linha por log.
- Campos obrigatórios presentes: `ts`, `level`, `logger`, `msg`, `module`, `func`, `line`, `pid`, `host`, `actor`, `actor_kind`.
- `ts` termina em `Z` (UTC).
- `host` é short (sem `.`).
- `service`/`service_version` aparecem quando setados em `setup_logging`; ausentes quando não.
- `execution_id` aparece após `set_execution_id("abc")`; some após `set_execution_id(None)`.
- `metadata` aninhado quando `logger.info("msg", extra={"k": "v"})`.
- `exc` estruturado em `logger.error(...)` dentro de `except`.
- `setup_logging` idempotente (segunda chamada não duplica handler).
- Env `JEDI_LOG_ACTOR` sobrescreve resolução por OS USER.
