# Tarefa — Implementação de `jedi_library.slug`

Funde `tili/dominio/slug.py::gerar_slug` e `jd-tasks/api/tag_utils.py::normalize_tag / normalize_tags`. Verificação empírica: 91 tags em produção do jd-tasks, 0 com espaço — fusão kebab é não-breaking.

## Entregáveis

### `jedi_library/slug.py`

```python
import re
import unicodedata


def normalize(text: str, *, max_len: int | None = None) -> str:
    """NFKD → ASCII → lower → strip → espaço→hífen → colapsa hífens.
    Input degenerado → "". Trunca no último hífen antes de max_len se fornecido."""
    nfkd = unicodedata.normalize("NFKD", text)
    ascii_text = nfkd.encode("ascii", "ignore").decode("ascii")
    lower = ascii_text.lower().strip()
    kebab = re.sub(r"[^a-z0-9-]+", "-", lower)
    kebab = re.sub(r"-+", "-", kebab).strip("-")

    if max_len is not None and len(kebab) > max_len:
        cortado = kebab[:max_len]
        ultimo = cortado.rfind("-")
        kebab = cortado[:ultimo].strip("-") if ultimo > 0 else cortado

    return kebab


def unique_slug(
    text: str,
    existing: set[str],
    *,
    max_len: int | None = None,
) -> str:
    """normalize(text, max_len=...) + sufixo "-N" (N≥2) se em existing.
    Input degenerado → ""."""
    base = normalize(text, max_len=max_len)
    if not base or base not in existing:
        return base
    n = 2
    while f"{base}-{n}" in existing:
        n += 1
    return f"{base}-{n}"


def normalize_unique(items: list[str]) -> list[str]:
    """[normalize(i) for i in items], descarta "", deduplica preservando ordem."""
    seen: set[str] = set()
    out: list[str] = []
    for item in items:
        n = normalize(item)
        if n and n not in seen:
            seen.add(n)
            out.append(n)
    return out
```

### Testes (`test/test_slug.py`)

```python
# normalize
assert normalize("Ação Rápida") == "acao-rapida"
assert normalize("  olá, mundo!  ") == "ola-mundo"
assert normalize("Meu-Projeto") == "meu-projeto"
assert normalize("Financeiro") == "financeiro"
assert normalize("℗") == ""
assert normalize("") == ""

# max_len
assert normalize("uma frase muito longa que precisa ser cortada", max_len=20) == "uma-frase-muito"

# unique_slug
assert unique_slug("tarefa", set()) == "tarefa"
assert unique_slug("tarefa", {"tarefa"}) == "tarefa-2"
assert unique_slug("tarefa", {"tarefa", "tarefa-2"}) == "tarefa-3"
assert unique_slug("℗", set()) == ""   # input degenerado → "" (sem raise)
assert unique_slug("℗", {""}) == ""

# normalize_unique
assert normalize_unique(["Financeiro", "FINANCEIRO", "Farmácia", "financeiro"]) == ["financeiro", "farmacia"]
assert normalize_unique(["Financeiro", "", "  ", "℗"]) == ["financeiro"]
assert normalize_unique([]) == []
```

### Compatibilidade verificada contra produção

Todas as 91 tags do banco de produção do `jd-tasks` passam por `normalize(tag) == tag` (idempotência).
