#### `open_connection(db_path, *, foreign_keys=True) -> sqlite3.Connection`

```python
import sqlite3
from pathlib import Path

def open_connection(
    db_path: str | Path,
    *,
    foreign_keys: bool = True,
) -> sqlite3.Connection:
    """Abre conexão SQLite com defaults da frota."""
    path_str = str(db_path)
    if path_str != ":memory:":
        Path(path_str).parent.mkdir(parents=True, exist_ok=True)

    conn = sqlite3.connect(path_str)
    conn.row_factory = sqlite3.Row
    if foreign_keys:
        conn.execute("PRAGMA foreign_keys = ON;")
    conn.execute("PRAGMA journal_mode = WAL;")
    return conn
```

#### `apply_migrations(conn, package, migrations_subdir="migrations") -> list[str]`

```python
import hashlib
import re
import sqlparse
from jedi_library.assets import list_files, read_text

_NOME_MIGRATION = re.compile(r"^(V\d+)__.+\.sql$")


def _ensure_schema_table(conn: sqlite3.Connection) -> None:
    conn.execute("""
        CREATE TABLE IF NOT EXISTS schema_migrations (
            versao      TEXT PRIMARY KEY,
            hash        TEXT NOT NULL,
            aplicada_em TEXT NOT NULL DEFAULT (datetime('now'))
        )
    """)
    conn.commit()


def _applied_versions(conn: sqlite3.Connection) -> dict[str, str]:
    """{versao: hash} das migrations já aplicadas."""
    cur = conn.execute("SELECT versao, hash FROM schema_migrations")
    return {row["versao"]: row["hash"] for row in cur.fetchall()}


def apply_migrations(
    conn: sqlite3.Connection,
    package: str,
    migrations_subdir: str = "migrations",
) -> list[str]:
    """Aplica migrations V\\d+__<descricao>.sql pendentes. Retorna versões aplicadas."""
    _ensure_schema_table(conn)
    applied = _applied_versions(conn)
    files = list_files(package, migrations_subdir, "V*.sql")

    # FK precisa estar fora de transação
    if conn.in_transaction:
        conn.commit()
    fk_originally_on = bool(conn.execute("PRAGMA foreign_keys").fetchone()[0])
    if fk_originally_on:
        conn.execute("PRAGMA foreign_keys = OFF;")

    aplicadas_agora: list[str] = []
    try:
        for f in files:
            m = _NOME_MIGRATION.match(f.name)
            if not m:
                continue  # arquivo fora do padrão — ignorado silenciosamente
            versao = m.group(1)
            sql = f.read_text(encoding="utf-8")
            digest = hashlib.sha256(sql.encode("utf-8")).hexdigest()

            if versao in applied:
                if applied[versao] != digest:
                    raise RuntimeError(
                        f"Migration {versao} ({f.name}) foi modificada após aplicação. "
                        f"Migrations aplicadas são imutáveis. "
                        f"Hash registrado: {applied[versao]}; hash atual: {digest}."
                    )
                continue

            sp = f"jedi_db_migration_{versao}"
            conn.execute(f"SAVEPOINT {sp};")
            try:
                for stmt in sqlparse.split(sql):
                    stmt = stmt.strip()
                    if stmt:
                        conn.execute(stmt)
                conn.execute(
                    "INSERT INTO schema_migrations (versao, hash) VALUES (?, ?)",
                    (versao, digest),
                )
                conn.execute(f"RELEASE SAVEPOINT {sp};")
                conn.commit()
            except Exception:
                conn.execute(f"ROLLBACK TO SAVEPOINT {sp};")
                conn.execute(f"RELEASE SAVEPOINT {sp};")
                conn.commit()
                raise
            aplicadas_agora.append(versao)

        violacoes = conn.execute("PRAGMA foreign_key_check;").fetchall()
        if violacoes:
            raise RuntimeError(
                f"Violações de FK após migrations: {[tuple(v) for v in violacoes]}"
            )
    finally:
        if fk_originally_on:
            if conn.in_transaction:
                conn.commit()
            conn.execute("PRAGMA foreign_keys = ON;")

    return aplicadas_agora
```

#### `transaction(conn)` context manager

```python
from contextlib import contextmanager
from typing import Iterator

@contextmanager
def transaction(conn: sqlite3.Connection) -> Iterator[sqlite3.Connection]:
    """commit em sucesso; rollback em exceção."""
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
```

### Testes (`test/test_db.py`)

- `open_connection(":memory:")` retorna conexão com FK=ON, row_factory=Row.
- `open_connection(tmp_path / "test.db")` cria diretório pai automaticamente.
- `apply_migrations` cria `schema_migrations` + aplica V*.sql em ordem.
- Idempotência: segunda chamada não re-aplica nada (retorna lista vazia).
- Hash mismatch em migration já aplicada → `RuntimeError`.
- Migration com erro: rollback do SAVEPOINT; `schema_migrations` não recebe registro.
- Arquivo fora do padrão `V\d+__*.sql` é ignorado sem erro.
- `PRAGMA foreign_key_check` rejeita ao final se migration deixar FK quebrada.
- Splitter robusto: SQL com `;` em string literal e em corpo de trigger funciona via `sqlparse`.
- `transaction(conn)` faz commit em sucesso e rollback em exceção.
